### 5.3 Pipeline

```python
# app/analyze_pipeline_v2.py

class RoadAnomalyPipeline:
    def __init__(self):
        self.motion_detector = MotionDetector()
        self.anomaly_identifier = AnomalyIdentifier()
        self.rag_service = RAGService()
        self.decision_service = DecisionService()

    async def run(self, video_frame) -> AlertResult:
        """完整流程"""
        # 1. 帧差法检测
        rois = self.motion_detector.detect(video_frame)

        # 2. 异常识别 (Gemma)
        for roi in rois:
            anomaly = self.anomaly_identifier.identify(roi)

            if anomaly.has_incident:
                # 3. RAG检索
                rag_context = self.rag_service.retrieve(
                    incident_type=anomaly.incident_type,
                    visual_evidence=anomaly.visual_evidence
                )

                # 4. 决策输出 (Gemma)
                decision = self.decision_service.decide(
                    anomaly=anomaly,
                    rag_context=rag_context
                )

                return AlertResult(**decision)

        return None  # 无异常
```
